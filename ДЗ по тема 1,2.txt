Тема 2. Задание 2:

DROP TABLE IF EXISTS catalogs;
CREATE TABLE catalogs (
id SERIAL PRIMARY KEY,
name VARCHAR (255) comment 'Название раздела'
) comment = 'Разделы хранилища медиа-файлов';

DROP TABLE IF EXISTS users;
CREATE TABLE users (
id SERIAL PRIMARY KEY,
name VARCHAR (255) comment 'Имя пользователя'
) comment = 'Пользователи';

DROP TABLE IF EXISTS files;
CREATE TABLE files (
id SERIAL PRIMARY KEY,
name VARCHAR (255) comment 'Имя файла',
discription TEXT comment 'Описание типа файла',
file_path VARCHAR (255) comment 'Путь к файлу',
users_id INT UNSIGNED,
catalogs_id INT UNSIGNED,
KEY index_of_users_id(users_id),
KEY index_of_catalogs_id(catalogs_id)
) comment = 'Хранимые файлы';
